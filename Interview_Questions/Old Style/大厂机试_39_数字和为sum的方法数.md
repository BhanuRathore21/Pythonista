# 递归关系式
# dp[i][j] = dp[i-1][j] + dp[i-1][j-A[i]]
# i表示数组下标，j表示前i个元素的和，dp[i][j]表示前i个元素和为j的方案数
n, m = map(int, input().split())
a = map(int, input().split())

dp = [0 for i in range(m + 1)]
dp[0] = 1

for i in range(n):
    j = m
    while j >= a[i]:
        dp[j] += dp[j - a[i]]
        j -= 1
print(dp[m])
