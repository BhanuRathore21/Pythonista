def longestpalindrome(s):
    if s == s[::-1]:
        return len(s)
    maxlen = 0
    for i in range(len(s)):
        if i - maxlen >= 1 and s[i - maxlen - 1:i + 1] == s[i - maxlen - 1:i + 1][::-1]:
            maxlen += 2
            print(s[i - maxlen - 1:i + 1], maxlen)

        if i - maxlen >= 0 and s[i - maxlen:i + 1] == s[i - maxlen:i + 1][::-1]:
            maxlen += 1
            print(s[i - maxlen - 1:i + 1], maxlen)
